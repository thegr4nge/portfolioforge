    class QualityFlag(IntFlag):
        """Bitmask flags for OHLCV data quality issues.

        A row with quality_flags == 0 is clean.
        The validator sets these flags after ingestion.
        Downstream backtest layer should filter or warn on non-zero flags.
        """
        ZERO_VOLUME       = 0x01  # Volume is zero on a day the exchange was open
        OHLC_VIOLATION    = 0x02  # low > open, close > high, or similar constraint broken
        PRICE_SPIKE       = 0x04  # Single-day move >50% with no corresponding split/dividend

        FX_ESTIMATED      = 0x10  # FX rate for this date was interpolated (no exact match)
        ADJUSTED_ESTIMATE = 0x20  # Adjustment factor estimated, not computed from exact data

  <action>
    Create `src/market_data/db/writer.py` with class `DatabaseWriter`:

    Constructor: `def __init__(self, conn: sqlite3.Connection) -> None` — stores conn as instance variable.

    Methods (all accept the relevant model types from db/models.py):

    **`upsert_security(self, record: SecurityRecord) -> int`**
    - INSERT INTO securities (ticker, name, exchange, sector, industry, currency, is_active, listed_date, delisted_date)
    - ON CONFLICT(ticker) DO UPDATE SET name=excluded.name, exchange=excluded.exchange, sector=excluded.sector, industry=excluded.industry, currency=excluded.currency, is_active=excluded.is_active
    - Return the security_id (use `cursor.lastrowid` or SELECT after upsert)

    **`upsert_ohlcv(self, records: list[OHLCVRecord]) -> int`**
    - Uses `executemany` for bulk insert
    - SQL: `INSERT INTO ohlcv (security_id, date, open, high, low, close, volume, adj_close, adj_factor, quality_flags) VALUES (?,?,?,?,?,?,?,?,?,0) ON CONFLICT(security_id, date) DO UPDATE SET open=excluded.open, high=excluded.high, low=excluded.low, close=excluded.close, volume=excluded.volume, adj_close=excluded.adj_close, adj_factor=excluded.adj_factor`
    - CRITICAL: `quality_flags` is set to 0 on INSERT but NOT touched in the DO UPDATE clause — the validator owns that column
    - Returns count of rows processed (len(records))
    - Commits inside a `with self.conn:` context manager

    **`upsert_dividends(self, records: list[DividendRecord]) -> int`**
    - ON CONFLICT(security_id, ex_date, dividend_type) DO UPDATE all fields except id
    - Includes franking fields: franking_credit_pct, franking_credit_amount, gross_amount

    **`upsert_splits(self, records: list[SplitRecord]) -> int`**
    - ON CONFLICT(security_id, ex_date) DO UPDATE split_from, split_to

    **`upsert_fx_rates(self, records: list[FXRateRecord]) -> int`**
    - ON CONFLICT(date, from_ccy, to_ccy) DO UPDATE rate=excluded.rate

    **`write_ingestion_log(self, record: IngestionLogRecord) -> None`**
    - Plain INSERT (no upsert — every log entry is a new record)
    - Never fails silently: if INSERT fails, let the exception propagate

    **`upsert_coverage(self, record: CoverageRecord) -> None`**
    - ON CONFLICT(security_id, data_type, source, from_date, to_date) DO UPDATE records=excluded.records, fetched_at=datetime('now')

    **`update_quality_flags(self, security_id: int, date: str, flags: int) -> None`**
    - Simple UPDATE ohlcv SET quality_flags=? WHERE security_id=? AND date=?
    - This is the ONLY method that touches quality_flags after initial insert

    All methods use loguru logger for debug-level logging. Type hints on all parameters and return types.
  </action>
  <verify>
    <automated>cd /home/hntr/market-data && source .venv/bin/activate && python -c "
import sqlite3
from src.market_data.db.schema import run_migrations
from src.market_data.db.writer import DatabaseWriter
from src.market_data.db.models import OHLCVRecord, SecurityRecord

conn = sqlite3.connect(':memory:')
run_migrations(conn)
w = DatabaseWriter(conn)

# Insert security
sec = SecurityRecord(ticker='AAPL', exchange='NASDAQ', currency='USD', name='Apple Inc')
sec_id = w.upsert_security(sec)
assert sec_id is not None

# Insert OHLCV
r = OHLCVRecord(security_id=sec_id, date='2024-01-02', open=185.0, high=186.0, low=184.0, close=185.5, volume=50000000, adj_close=185.5)
count = w.upsert_ohlcv([r])
assert count == 1

# Set quality flag
from src.market_data.quality.flags import QualityFlag
w.update_quality_flags(sec_id, '2024-01-02', int(QualityFlag.ZERO_VOLUME))
flags = conn.execute('SELECT quality_flags FROM ohlcv WHERE security_id=? AND date=?', (sec_id, '2024-01-02')).fetchone()[0]
assert flags == 0x01

# Re-upsert SAME row — quality_flags must NOT be reset
w.upsert_ohlcv([r])
flags_after = conn.execute('SELECT quality_flags FROM ohlcv WHERE security_id=? AND date=?', (sec_id, '2024-01-02')).fetchone()[0]
assert flags_after == 0x01, f'quality_flags was reset! got {flags_after}'
print('writer OK')
